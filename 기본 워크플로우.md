# 기본 워크플로우

## 저장소 생성 및 관리

```bash
# 디렉터리 생성 및 이동
mkdir git
cd git

# 현재 폴더를 Git 저장소로 초기화 (.git 생성)
git init

# Git 저장소 삭제 (버전 관리 중단)
rm -rf .git

## 파일 작업 및 커밋

# 1. 파일 생성 및 수정 (Working Directory)
echo hello world! > a.txt

# 2. 스테이징 영역에 추가 (Staging Area)
git add *.txt

# 3. 상태 확인
git status      # 자세히 보기
git status -s   # 요약 보기

# 4. 스테이징 취소 (필요시)
git rm --cached *

# 5. 커밋하기 (Repository 저장)
git commit

# 6. 커밋 이력 확인
git log
